5/// Window handle.

9impl Window {
10    /// Move Window to (x, y) relative the screen.
11    ///
12    /// # Parameters
13    /// * `x` - The x coordinate of the new position.
14    /// * `y` - The y coordinate of the new position.
15    /// * `redraw` - If true, the window will be redrawn.
16    pub fn move_to(&mut self, x: i32, y: i32, redraw: bool) {
17        unsafe { ege_movewindow(x, y, redraw) };
18    }
19
20    /// Resize the window to (width, height).
21    ///
22    /// # Parameters
23    /// * `width` - The new width of the window.
24    /// * `height` - The new height of the window.
25    pub fn resize(&mut self, width: i32, height: i32) {
26        unsafe { ege_resizewindow(width, height) };
27    }
28
29    /// Flushes the window.
30    pub fn flush(&mut self) {
31        unsafe { ege_flushwindow() };
32    }
33
34    /// Show the window.
35    ///
36    /// # Note
37    /// If the initialization mode contains `Init::Hide`,
38    /// then this function needs to be called for the window to display.
39    pub fn show(&mut self) {
40        unsafe { ege_showwindow() };
41    }
42
43    /// Hide the window.
44    pub fn hide(&mut self) {
45        unsafe { ege_hidewindow() };
46    }

48    /// Get the state of specified key.
49    ///
50    /// # Parameters
51    /// * `key` - The key to check.
52    ///
53    /// # Returns
54    /// * `true` if the key is pressed, `false` otherwise.
55    pub fn keystate(&self, key: Key) -> bool {
56        unsafe { ege_keystate(<Key as Into<u32>>::into(key) as i32) }
57    }
58
59    /// Get next character message.
60    /// 
61    /// # Returns
62    /// * `Some(char)` if has character message, `None` otherwise.
63    pub fn getchar(&self) -> Option<char> {
64        if unsafe { ege_kbhit() != 0 } {
65            char::from_u32(unsafe { ege_getch() as u32 })
66        } else {
67            None
68        }
69    }
70
71    /// Get next key message.
72    ///
73    /// # Returns
74    /// * `Some((KeyMsg, KeyFlags))` if has key message, `None` otherwise.
75    pub fn getmsg(&self) -> Option<(KeyMsg, KeyFlags)> {
76        if unsafe { ege_kbmsg() != 0 } {
77            let msg = unsafe { ege_getkey() };
78            let keymsg = match msg.msg {
79                xege_ffi::ege_key_msg_e_key_msg_down => KeyMsg::Down(Key::from(msg.key as u32)),
80                xege_ffi::ege_key_msg_e_key_msg_up => KeyMsg::Up(Key::from(msg.key as u32)),
81                xege_ffi::ege_key_msg_e_key_msg_char => {
82                    KeyMsg::Char(char::from_u32(msg.key as u32)?)
83                }
84                _ => unreachable!(),
85            };
86            let mut flags = KeyFlags::none();
87            if (msg.flags & xege_ffi::ege_key_flag_e_key_flag_shift as u32) != 0 {
88                flags |= KeyFlags::Shift;
89            }
90            if (msg.flags & xege_ffi::ege_key_flag_e_key_flag_ctrl as u32) != 0 {
91                flags |= KeyFlags::Ctrl;
92            }
93            if (msg.flags & xege_ffi::ege_key_flag_e_key_flag_first_down as u32) != 0 {
94                flags |= KeyFlags::First;
95            }
96            Some((keymsg, flags))
97        } else {
98            None
99        }
100    }
101
102    /// Flush the key message buffer.
103    pub fn flushkey(&self) {
104        unsafe { ege_flushkey() };
105    }
106
107    /// Get next mouse message.
108    ///
109    /// # Returns
110    /// * `Some(MouseMsg)` if has mouse message, `None` otherwise.
111    pub fn getmouse(&self) -> Option<MouseMsg> {
112        if unsafe { ege_mousemsg() != 0 } {
113            let msg = unsafe { ege_getmouse() };
114            Some(MouseMsg { msg })
115        } else {
116            None
117        }
118    }

120    /// Get the current mouse position.
121    /// 
122    /// # Returns
123    /// The current mouse position.
124    pub fn mousepos(&self) -> Point {
125        let mut p = Point { x: 0, y: 0 };
126        unsafe { ege_mousepos(&mut p.x, &mut p.y) };
127        p
128    }

132    /// # Parameters
133    /// * `visible` - If true, the mouse cursor will be visible.

135        unsafe { ege_showmouse(show as i32) };
136    }
