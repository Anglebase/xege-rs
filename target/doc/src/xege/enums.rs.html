392}
393
394impl Default for PitchAndFamily {
395    fn default() -> Self {
396        PitchAndFamily {
397            mask: DEFAULT_PITCH | (FF_DONTCARE as u32),
398        }
399    }
400}

402impl BitOr<Pitch> for FontFamily {
403    type Output = PitchAndFamily;
404
405    fn bitor(self, rhs: Pitch) -> Self::Output {
406        PitchAndFamily {
407            mask: (self as u32) | (rhs as u32),
408        }
409    }
410}
411
412impl BitOr<FontFamily> for Pitch {
413    type Output = PitchAndFamily;
414
415    fn bitor(self, rhs: FontFamily) -> Self::Output {
416        rhs | self
417    }
418}
419
420impl TryFrom<u32> for PitchAndFamily {
421    type Error = ();
422
423    fn try_from(value: u32) -> Result<Self, Self::Error> {
424        let pitch = Pitch::try_from(value & 0x00000003)?;
425        let family = FontFamily::try_from((value & 0x000000F0) >> 4)?;
426        Ok(PitchAndFamily {
427            mask: (pitch as u32) | ((family as u32) << 4),
428        })
429    }
430}

450impl Into<LOGFONTW> for Font {
451    fn into(self) -> LOGFONTW {
452        let mut buff = [0u16; 32];
453        let facename = self
454            .facename
455            .encode_utf16()
456            .chain(Some(0))
457            .collect::<Vec<u16>>();
458        for (i, &c) in facename.iter().enumerate() {
459            buff[i] = c;
460        }
461        LOGFONTW {
462            lfHeight: self.height,
463            lfWidth: self.width,
464            lfEscapement: self.escapement,
465            lfOrientation: self.orientation,
466            lfWeight: self.weight as i32,
467            lfItalic: self.italic as u8,
468            lfUnderline: self.underline as u8,
469            lfStrikeOut: self.strikeout as u8,
470            lfCharSet: self.charset as u8,
471            lfOutPrecision: self.outprecision as u8,
472            lfClipPrecision: self.clipprecision as u8,
473            lfQuality: self.quality as u8,
474            lfPitchAndFamily: self.pitchandfamily.mask as u8,
475            lfFaceName: buff,
476        }
477    }
478}
479
480impl TryFrom<LOGFONTW> for Font {
481    type Error = ();
482
483    fn try_from(value: LOGFONTW) -> Result<Self, Self::Error> {
484        let facename = String::from_utf16(&value.lfFaceName).map_err(|_| ())?;
485        Ok(Font {
486            height: value.lfHeight,
487            width: value.lfWidth,
488            escapement: value.lfEscapement,
489            orientation: value.lfOrientation,
490            weight: Weight::try_from(value.lfWeight)?,
491            italic: value.lfItalic != 0,
492            underline: value.lfUnderline != 0,
493            strikeout: value.lfStrikeOut != 0,
494            charset: CharSet::try_from(value.lfCharSet as u32)?,
495            outprecision: OutPrecision::try_from(value.lfOutPrecision as u32)?,
496            clipprecision: ClipPrecision::try_from(value.lfClipPrecision as u32)?,
497            quality: Quality::try_from(value.lfQuality as u32)?,
498            pitchandfamily: PitchAndFamily::try_from(value.lfPitchAndFamily as u32)?,
499            facename,
500        })
501    }
502}
